# print fields in a csv line and total fields processed (which will be one less than the actual number of records - that is, starts from zero)
echo $opts| awk 'BEGIN {FS=","} {print $1 " " $2 " " $3} END {print NR, " fields processed"}'

# print number of fields (which will be one less than the actual number of records - that is, starts from zero)
echo $opts| awk 'BEGIN {FS=","} {print NR}'
